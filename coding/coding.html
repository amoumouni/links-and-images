Sample Snippet

Small utility: run a function once per session using sessionStorage. 

// Run a function once per session
function oncePerSession(key, fn){
  try {
    if (!sessionStorage.getItem(key)) {
      fn();
      sessionStorage.setItem(key, '1');
    }
  } catch(e) { fn(); }
}
